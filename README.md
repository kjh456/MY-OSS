-- 주소록 테이블 생성

    이름        char(10)           NOT NULL ,
    전화번호  char(13) ,
    주소        varchar(10) ,
    생일        varchar(11) ,
    PRIMARY KEY(이름)
);

-- 주소록(이름, 전화번호, 주소, 생일)

INSERT INTO 주소록

select * from 주소록;
